```bash

# PVC 상태 확인
kubectl get pvc -n shop-msa

# PV 확인
kubectl get pv

# StorageClass 확인
kubectl get storageclass

# Pod 상태 모니터링
kubectl get pods -n shop-msa -w

# Kafka 에러 로그 확인
kubectl logs -f kafka-0 -n shop-msa

# 이벤트 로그 확인
kubectl get events -n shop-msa

# Pod 상태 상세 확인
kubectl describe pod kafka-0 -n shop-msa | tail -50

# ruok (Are you OK?)
kubectl exec zookeeper-0 -n shop-msa -- bash -c "echo ruok | nc localhost 2181"
# 출력: imok

# mntr (모니터링 정보)
kubectl exec zookeeper-0 -n shop-msa -- bash -c "echo mntr | nc localhost 2181"

# conf (설정 정보)
kubectl exec zookeeper-0 -n shop-msa -- bash -c "echo conf | nc localhost 2181"

# redis-master HOSTNAME 확인
kubectl exec redis-master-0 -n shop-msa -- hostname -f

```
